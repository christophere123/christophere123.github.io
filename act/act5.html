Pantalla Oled






en la oled tu podras mostrar la informacion requerida, incluso datos relevantes de las raspberry pi 3

pip install adafruit-circuitpython-ssd1306


import board
import busio
import adafruit_ssd1306
from PIL import Image, ImageDraw, ImageFont

# Configuración del OLED
WIDTH = 128
HEIGHT = 32
I2C_ADDR = 0x3C

# Inicializar el objeto I2C
i2c = busio.I2C(board.SCL, board.SDA)

# Inicializar el objeto OLED
oled = adafruit_ssd1306.SSD1306_I2C(WIDTH, HEIGHT, i2c, addr=I2C_ADDR)

# Limpiar la pantalla
oled.fill(0)
oled.show()

# Crear una imagen en blanco
image = Image.new("1", (WIDTH, HEIGHT))
draw = ImageDraw.Draw(image)

# Cargar una fuente
font = ImageFont.load_default()

# Escribir "Hola Mundo" en la imagen
draw.text((0, 0), "Hola Mundo", font=font, fill=255)

# Mostrar la imagen en el OLED
oled.image(image)
oled.show()
