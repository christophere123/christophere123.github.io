Buzzer






El buzzer es capaz de reproducir melodias sencillas mediante vibraciones, las mas comunes son las melodias navideñas

import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM)
GPIO.setup(17, GPIO.OUT)

melodia = [
261, 293, 329, 349, 392, 440, 493, 523
]

duraciones = [
0.5, 0.5, 0.5, 0.5, 0.5, 0.5, 0.5, 0.5
]

def reproducir_melodia(pin, melodia, duraciones):
for nota, duracion in zip(melodia, duraciones):
generar_tono(pin, nota, duracion)
time.sleep(0.1) # Pausa entre notas

def generar_tono(pin, frecuencia, duracion):
pwm = GPIO.PWM(pin, frecuencia)
pwm.start(50) # Inicia la señal PWM con un ciclo de trabajo del 50% (50% de volumen)
time.sleep(duracion)
pwm.stop()

try:
reproducir_melodia(17, melodia, duraciones)

except KeyboardInterrupt:
GPIO.cleanup()
