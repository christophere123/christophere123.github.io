Sensor touch






Cuando el sensor detecta el contacto este manda la señal del contacto detectado, la sensibilidad debe ser ajustada manualmente

import os
import time
time.sleep(0.100)
import random

import RPi.GPIO as GPIO

GPIO.setmode(GPIO.BCM)

GPIO.setup(9, GPIO.IN, pull_up_down=GPIO.PUD_UP)

while True:
if GPIO.input(9) == True:
print("contacto detectado")
time.sleep(0.5)

if GPIO.input(9) == False:
print("sin contacto")
time.sleep(0.5)
