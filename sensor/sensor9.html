Rotary Encoder






Este sensor se encarga de enviar señales de acuerdo al giro de este mismo, puede funcionar trabajando con 2 casos, falso y verdadero que seria como este ejemplo, aunque tambien podria trabajar con numeros

import RPi.GPIO as GPIO
import time

SW_PIN = 20
DT_PIN = 16
CLK_PIN = 17

GPIO.setmode(GPIO.BCM)
GPIO.setup(SW_PIN, GPIO.IN, pull_up_down=GPIO.PUD_UP)
GPIO.setup(DT_PIN, GPIO.IN)
GPIO.setup(CLK_PIN, GPIO.IN)

valor_anterior = True
switch_presionado = False

while True:
if valor_anterior != GPIO.input(CLK_PIN):
if GPIO.input(CLK_PIN) == False:
if GPIO.input(DT_PIN) == False:
print("izquierda")
time.sleep(0.2)
else:
print("derecha")
time.sleep(0.2)
valor_anterior = GPIO.input(CLK_PIN)

if GPIO.input(SW_PIN) == False and not switch_presionado:
print("SW presionado")
switch_presionado = True
if GPIO.input(SW_PIN) == True and switch_presionado:
switch_presionado = False
